\begin{frame}[fragile]{The New Zealand Attitudes and Values Study}
\protect\hypertarget{the-new-zealand-attitudes-and-values-study}{}
\begin{itemize}[<+->]
\tightlist
\item
  Planned 20-year longitudinal study, currently in its 12\(^{th}\) year.
\item
  Sample frame drawn randomly from NZ Electoral Roll.
\item
  Postal questionnaire (coverage; retention \textasciitilde{} 80\%)
\item
  Focus on personality, social attitudes, values, religion,
  meaning-making, perfectionism, virtues, habits, employment,
  experiences of discrimination, physical and psychological health, and
  environmental attitudes \ldots{}
\item
  Current sample contains \texttt{36524} unique people, which is \(>\)
  1\% of the NZ adult population.
\item
  Here, \emph{N} = 34778 longitudinal participants who responded to
  \textbf{both} the 2018/19 and 2019/20 NZAVS waves.
\end{itemize}
\end{frame}
